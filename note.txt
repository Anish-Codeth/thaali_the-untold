//for user credentials
create table vendor_credentials(

    email text not null primary key,
    password text not null
)

//for vendor profile
drop table if exists vendor_profile;

create table vendor_profile(
email varchar(255) references vendor_credentials(email) on delete cascade,
name varchar(255) not null ,
phone_number bigint not null,
photos text,
description text,
booking_status bool,
username int primary key,//dont forget to change
latitude numeric,
longitude numeric  --will be in squared form
)
